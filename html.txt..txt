<script>
/* ------------------------------
   1. MUSIC PLAYLIST
--------------------------------*/
const playlist = [
  "song1.mp3",
  "song2.mp3",
  "song3.mp3"
];

const audio = document.getElementById("musicPlayer");

// shuffle and play
function playRandomSong() {
  audio.src = playlist[Math.floor(Math.random() * playlist.length)];
  audio.play();
}

// autoplay next random song
audio.onended = playRandomSong;

playRandomSong(); // start music


/* ------------------------------
   2. IMAGE SLIDESHOW
--------------------------------*/
const images = [
  "photo1.jpg",
  "photo2.jpg",
  "photo3.jpg",
  "photo4.jpg",
  "photo5.jpg"
];

let index = 0;
let running = true;
let timer = null;

const slideshow = document.getElementById("slideshow");

// fade-in image
function showImage() {
  slideshow.style.opacity = 0;
  setTimeout(() => {
    slideshow.src = images[index];
    slideshow.style.opacity = 1;
  }, 400);
}

// auto-advance
function nextImage() {
  index = (index + 1) % images.length;
  showImage();
}

function startShow() {
  if (running) return;
  running = true;
  timer = setInterval(nextImage, 5000);
}

function stopShow() {
  running = false;
  clearInterval(timer);
}

// start slideshow automatically
timer = setInterval(nextImage, 5000);
showImage();

</script>
